// 1. Define the specific Rotterdam-area bounding box // Using a smaller, precise area keeps the response small const rotterdamBounds = 'POLYGON((4.3 51.8, 4.6 51.8, 4.6 52.0, 4.3 52.0, 4.3 51.8))'; // 2. Build the query to filter by location AND type (Temperature) // This keeps the response size very small const baseUrl = 'https://api-virtualair.nilu.no/v1.1/Things'; const query = `?$expand=Datastreams/Observations($orderby=phenomenonTime%20desc;$top=1),Locations` + `&$filter=st_within(Locations/location, geography'${rotterdamBounds}')` + ` and Datastreams/ObservedProperty/definition eq 'http://vocabs.lter-europe.net/EnvThes/22035'`; const apiUrl = 'https://corsproxy.io/?url=' + encodeURIComponent(baseUrl + query); // 3. Fetch with confidence fetch(apiUrl) .then(response => { if (!response.ok) throw new Error('API request failed'); return response.json(); }) .then(data => { console.log("Filtered Data:", data); const heatPoints = []; // Ensure we handle the standard SensorThings 'value' array const sensors = data.value || []; sensors.forEach(thing => { const coords = thing.Locations?.[0]?.location?.coordinates; const temp = thing.Datastreams?.[0]?.Observations?.[0]?.result; if (coords && temp !== null && temp !== undefined) { // Leaflet uses [lat, lon, intensity] heatPoints.push([coords[1], coords[0], temp]); } }); if (heatPoints.length > 0) { L.heatLayer(heatPoints, {radius: 30, blur: 20}).addTo(map); console.log("Mapped " + heatPoints.length + " sensors."); } else { console.warn("No sensors found in Rotterdam with this filter."); } }) .catch(err => console.error("Error loading data:", err));
